89  * The standard deviation of a sequence of numbers is can be computed

94 double StatsCalculator::computeStandardDeviation(){
95 
96  // If any numeric values were successfully parsed from the input file...
97  if(numericValues.size() > 0){
98 
99  /* Declare and zero-initialize a double precision variable with
100  * identifier "meanOfSquaredValues" that will ultimately contain
101  * the mean of the squares of the elements of the "numericValues"
102  * member datum.
103  */
104  double meanOfSquaredValues(0.0);
105 
106  /* Use the range-based for-loop syntax to iterate over all elements
107  * of the "numericValues" member datum.
108  *
109  * The value of the current element of "numericValues" will be
110  * accessible within the for-loop body using the identifier
111  * "numericValue".
112  */
113  for(double numericValue : numericValues){
114  /* Update the value of "meanOfSquaredValues" by adding the square
115  * of the value of the current element of "numericValues"
116  */
117  meanOfSquaredValues += numericValue*numericValue;
118  }
119 
120  /* The current value of "meanOfSquaredValues" corresponds to the SUM
121  * of the squares of the elements of "numericValues". The required mean
122  * can be obtained by dividing by the number of elements comprising
123  * numericValues. This number is returned by the size() method
124  * of the std::vector<double>-type member datum "numericValues".
125  */
126  meanOfSquaredValues /= numericValues.size();
127 
128  /* Compute and return the required standard deviation. The square
129  * of the mean of the elements of "numericValues" is computed
130  * by delegating computation of the mean to getMean() and squaring
131  * the result.
132  *
133  * Technical note: It would be more efficient to call getMean() once,
134  * then store and square the result. In reality, almost all modern
135  * optimizing compilers will recognize this and produce equivalent
136  * machine code from the subsequent statement.
137  *
138  * The STL function std::sqrt provided by the <cmath> header file is
139  * used to compute the square root.
140  */
141  return std::sqrt(meanOfSquaredValues - getMean()*getMean());
142  }
143  else{ // No numeric values were successfully parsed from the input file
144  return 0.0;

248  * to the is_open() method provided by std::ifstream returns
249  * true. The latter criterion is established if the good()
250  * method provided by std::ifstream returns true.
251  *
252  * Technical Note: In C++ the logical and "&&" operator uses
253  * "short-circuit" semantics. Since both operands MUST evaluate
254  * to true in order for the expression to be true, it is possible
255  * to determine the expression result by examining only one
256  * operand IF that operand evaluates to false. By convention
257  * C++ examines the left-hand operand first and does not evaluate

267  * inputFile.eof().
268  */
269  while(!inputFile.eof() && inputFile.good()){
270  /* The stream input operator (">>") extracts a whitespace-

278  * invoked to append the extracted numeric value to the end of

282  }
283  /* Explicitly close the input file, freeing any resources it acquired
284  * when it was constructed or during its operation.
285  */
286  inputFile.close();
287  }
